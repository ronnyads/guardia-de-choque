```typescript
// ADICIONAR imports
import { getStoreConfig } from '@/lib/store-config';
import { TenantProvider } from '@/components/providers/TenantProvider';

// SUBSTITUIR export const metadata por:
export async function generateMetadata() {
  const config = await getStoreConfig();
  return {
    title: config.seo_title ?? 'Guardiã de Choque | Defesa Pessoal',
    description: config.seo_description ?? 'A defesa pessoal que sua família merece.',
    // manter openGraph e outros campos existentes, usando config quando disponível
  };
}

// TORNAR RootLayout async e buscar config:
export default async function RootLayout({ children }: { children: React.ReactNode }) {
  const config = await getStoreConfig();

  return (
    <html
      lang="pt-BR"
      style={{
        '--store-primary': config.primary_color,
        '--store-accent': config.accent_color,
      } as React.CSSProperties}
      // manter className existente com font variables
    >
      <body>
        {/* manter todos os providers e componentes existentes */}
        <TenantProvider config={config}>
          {children}
        </TenantProvider>
      </body>
    </html>
  );
}
```
