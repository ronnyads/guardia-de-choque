## TASK 3 — Criar src/app/admin/products/new/actions.ts (Server Action)

**Arquivo novo.** O formulário de criação de produto usa Server Action para garantir que tenant_id seja injetado server-side (nunca pelo client):

```typescript
'use server';

import { requireTenant } from '@/lib/tenant';
import { createServerSupabase } from '@/lib/supabase-server';
import { redirect } from 'next/navigation';
import { revalidatePath } from 'next/cache';

export async function createProduct(formData: FormData) {
  const { tenantId } = await requireTenant();
  const supabase = await createServerSupabase();

  const name = formData.get('name') as string;
  const slug = formData.get('slug') as string;
  const status = (formData.get('status') as string) || 'draft';
  const original_price = parseFloat(formData.get('original_price') as string || '0');
  const promo_price = parseFloat(formData.get('promo_price') as string || '0');
  const sku = formData.get('sku') as string;
  const inventory_count = parseInt(formData.get('inventory_count') as string || '0');
  const description = formData.get('description') as string;

  if (!name || !slug) {
    throw new Error('Nome e slug são obrigatórios');
  }

  const { error } = await supabase.from('products').insert({
    tenant_id: tenantId,
    name,
    slug,
    status,
    original_price,
    promo_price,
    sku,
    inventory_count,
    description,
  });

  if (error) {
    throw new Error(`Erro ao criar produto: ${error.message}`);
  }

  revalidatePath('/admin/products');
  redirect('/admin/products');
}
```
