  ```typescript
  // Apos os campos existentes, adicionar:
  const long_description = (formData.get('long_description') as string) || '';
  const badge = (formData.get('badge') as string) || null;
  const category_id = (formData.get('category_id') as string) || null;

  // Images: uma URL por linha, filtrar vazios
  const imagesRaw = (formData.get('images') as string) || '';
  const images = imagesRaw.split('\n').map(s => s.trim()).filter(Boolean);

  // Features: parse JSON com fallback para array vazio
  let features: unknown[] = [];
  try {
    const featuresRaw = (formData.get('features') as string) || '[]';
    const parsed = JSON.parse(featuresRaw);
    if (Array.isArray(parsed)) features = parsed;
  } catch {
    // JSON invalido — manter array vazio (nao quebrar)
  }

  // Specs: parse JSON com fallback para array vazio
  let specs: unknown[] = [];
  try {
    const specsRaw = (formData.get('specs') as string) || '[]';
    const parsed = JSON.parse(specsRaw);
    if (Array.isArray(parsed)) specs = parsed;
  } catch {
    // JSON invalido — manter array vazio
  }
  ```
